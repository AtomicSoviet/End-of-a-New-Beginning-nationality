# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = GetDissentLevel

	text = { # Content
		trigger = {
			AND = {
				check_variable = { nationality_dissent_level < 20 }
			}
		}
		localization_key = var_nationality_dissent_level_0
	}

	text = { # Disruptive
		trigger = {
			AND = {
				OR = {
					check_variable = { nationality_dissent_level > 20 }
					check_variable = { nationality_dissent_level = 20 }
				}
				check_variable = { nationality_dissent_level < 40 }
			}
		}
		localization_key = var_nationality_dissent_level_1
	}

	text = { # Restless
		trigger = {
			AND = {
				OR = {
					check_variable = { nationality_dissent_level > 40 }
					check_variable = { nationality_dissent_level = 40 }
				}
				check_variable = { nationality_dissent_level < 60 }
			}
		}
		localization_key = var_nationality_dissent_level_2
	}

	text = { # Agitated
		trigger = {
			AND = {
				OR = {
					check_variable = { nationality_dissent_level > 60 }
					check_variable = { nationality_dissent_level = 60 }
				}
				check_variable = { nationality_dissent_level < 80 }
			}
		}
		localization_key = var_nationality_dissent_level_3
	}

	text = { # Hostile
		trigger = {
			AND = {
				OR = {
					check_variable = { nationality_dissent_level > 80 }
					check_variable = { nationality_dissent_level = 80 }
				}
			}
		}
		localization_key = var_nationality_dissent_level_4
	}
}
