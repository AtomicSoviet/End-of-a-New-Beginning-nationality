on_actions = {
	on_state_control_changed = { #ROOT is new controller #FROM is old controller #FROM.FROM is state ID
		effect = {
			if = {
				limit = {
					AND = {
						check_variable = {
							FROM.FROM.nationality_group = ROOT.nationality_group
						}
						check_variable = {
							FROM.FROM.nationality = ROOT.nationality
						}
					}
				}
				subtract_from_variable = {
					var = FROM.FROM.nationality_dissent_level 
					value = 500
				}
			}

			if = {
				limit = {
					AND = {
						check_variable = {
							FROM.FROM.nationality_group = ROOT.nationality_group
						}
						NOT = {
							check_variable = {
								FROM.FROM.nationality = ROOT.nationality
							}
						}
					}
				}
				subtract_from_variable = {
					var = FROM.FROM.nationality_dissent_level 
					value = 100
				}
			}

			if = {
				limit = {
					AND = {
						NOT = {
							check_variable = {
								FROM.FROM.nationality_group = ROOT.nationality_group
							}
						}
						NOT = {
							check_variable = {
								FROM.FROM.nationality = ROOT.nationality
							}
						}
					}
				}
				add_to_variable = {
					var = FROM.FROM.nationality_dissent_level 
					value = 250
				}
			}
		}
	}
}
